## Command

```bash
HF_HOME=data/hf_cache HF_HUB_ENABLE_HF_TRANSFER=1 HF_HUB_DISABLE_XET=1 \
  .uv-bootstrap/bin/uv run python scripts/download_model.py \
  --model Qwen/Qwen3-32B \
  --max-storage 10TB

HIP_VISIBLE_DEVICES=0 scripts/rocm_docker_run.sh python scripts/evaluate_niah.py \
  --model Qwen/Qwen3-32B \
  --context-lengths 4096,8192,16384,32768,40000 \
  --depths 0,25,50,75,100 \
  --trials 3 \
```
